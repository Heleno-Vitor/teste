Comandos utilizados:

git config --global user.name "Heleno Vitor" //nome de usuário

git config --global user.email "helenovictormecatronica@gmail.com" //e-mail

git init  //inicia o repositório local

git status  //verifica alterações no repositório local

git add   //adiciona arquivos ao staging

git commit  //salva as alterações no histórico

git log  //verifica alterações no histórico

git remote add origin  //informa o repositório remoto

git remote -v //visualiza o repositório remoto

git push -u origin master  //publica as alterações no repositório remoto

git pull  //baixa as alterações do repositório remoto para o local

git checkout -b  //cria uma branch

git merge  //unifica branchs
